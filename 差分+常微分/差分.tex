\begin{equation}

\begin{equation}
u(n+2)=21u(n+2)+44(1+u(1)+u(2)+\cdots+u(n))
\end{equation}

（2）式减去（1）式得
\begin{equation}
u(n+2)-22u(n+1)-23u(n)=0
\end{equation}

由（3）式解得：
\begin{equation*}
u(n)=\frac{1}{24}[u(0)+u(1)]23^{n}+\frac{1}{24}[23u(0)-u(1)](-1)^{n}
\end{equation*}
